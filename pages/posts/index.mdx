    <script>
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const targetId = this.getAttribute('href');
                if (targetId === '#') return;
                const targetElement = document.querySelector(targetId);
                if (!targetElement) return;
                window.scrollTo({
                    top: targetElement.offsetTop - 80, 
                    behavior: 'smooth'
                });
            });
        });

        window.addEventListener('scroll', () => {
            const scrollY = window.scrollY;
            const scrollPercent = (scrollY / (document.body.scrollHeight - window.innerHeight)) * 100;
            document.body.style.setProperty('--scroll-position', `${scrollPercent}%`);
            document.querySelector('.blur-circle-1').style.transform = `translate(${scrollY * 0.05}px, ${scrollY * 0.02}px)`;
            document.querySelector('.blur-circle-2').style.transform = `translate(${-scrollY * 0.03}px, ${scrollY * 0.04}px)`;
            document.querySelector('.blur-circle-3').style.transform = `translate(${scrollY * 0.02}px, ${-scrollY * 0.03}px)`;
            document.querySelector('.blur-circle-4').style.transform = `translate(${-scrollY * 0.04}px, ${scrollY * 0.05}px)`;
            const nav = document.querySelector('nav');
            if (scrollY > 50) {
                nav.classList.add('scrolled');
            } else {
                nav.classList.remove('scrolled');
            }
        });
    </script>
